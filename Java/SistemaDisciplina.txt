package exercicios2;

public class Disciplina {
    String nome;
    String cod;

    public Disciplina(String nome, String cod) {
        this.nome = nome;
        this.cod = cod;
    }

    public String getNome() {
        return this.nome;
    }

    public void setNome(String nome) {
        this.nome = nome;
    }

    public String getCod() {
        return this.cod;
    }

    public void setCod(String cod) {
        this.cod = cod;
    }
    
    
}
