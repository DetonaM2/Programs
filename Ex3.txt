#include <stdio.h>
#include <stdlib.h>
#include <locale.h>

void asterisco(int n){
	int i;
	for(i=0;i<n;i++){
		printf("*");
	}
	return;
}

	setlocale(LC_ALL,"Portuguese");
	int n;

	scanf("%d", &n);

	printf("\nNÃºmeros de 1 a 5\n");
	asterisco(n);
	printf("\n1\n2\n3\n4\n5\n");
	asterisco(n);
	return 0;
}
